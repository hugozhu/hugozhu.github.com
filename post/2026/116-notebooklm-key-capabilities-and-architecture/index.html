  • YouTubeи§Ҷйў‘пјҡжҸҗеҸ–еӯ—е№•е’Ңи§Ҷи§үдҝЎжҒҜ

иҝҷдёҚд»…д»…жҳҜз®ҖеҚ•зҡ„ж–Үжң¬жҸҗеҸ–пјҢиҖҢжҳҜж·ұеәҰиҜӯд№үзҗҶи§Јпјҡ

  • иҜҶеҲ«ж–ҮжЎЈз»“жһ„пјҲз« иҠӮгҖҒж ҮйўҳгҖҒеҲ—иЎЁпјү

  • жҸҗеҸ–е…ій”®и®әзӮ№е’Ңж”Ҝж’‘иҜҒжҚ®
  • дҝқз•ҷдёҠдёӢж–Үе’Ңеј•з”Ёе…ізі»

2. еҹәдәҺжәҗж–ҮжЎЈзҡ„й—®зӯ”зі»з»ҹ

NotebookLMзҡ„дёҖдёӘжҳҫи‘—зү№зӮ№жҳҜжәҗеј•з”Ёзҡ„еҮҶзЎ®жҖ§гҖӮеҪ“е®ғеӣһзӯ”й—®йўҳж—¶пјҡ

  • дёҚдә§з”ҹе№»и§үпјҡжүҖжңүзӯ”жЎҲеҝ…йЎ»жңүжәҗж–ҮжЎЈж”Ҝж’‘
  • жҸҗдҫӣеј•з”ЁпјҡжҜҸдёӘеӣһзӯ”йғҪж ҮжіЁжқҘжәҗдҪҚзҪ®
  • еӨҡж–ҮжЎЈз»јеҗҲпјҡиғҪи·ЁеӨҡдёӘжәҗж–ҮжЎЈж•ҙеҗҲдҝЎжҒҜ
  • иҜӯеўғж„ҹзҹҘпјҡзҗҶи§Јй—®йўҳзҡ„йҡҗеҗ«ж„Ҹеӣҫ

иҝҷдёҺжҷ®йҖҡзҡ„LLMиҒҠеӨ©жңәеҷЁдәәеҪўжҲҗйІңжҳҺеҜ№жҜ”вҖ”вҖ”е®ғжҳҜдёҖдёӘеҸ—зәҰжқҹзҡ„гҖҒеҸҜйӘҢиҜҒзҡ„AIеҠ©жүӢгҖӮ

3. Audio OverviewпјҡиҮӘеҠЁз”ҹжҲҗж’ӯе®ў

иҝҷжҳҜNotebookLMжңҖд»ӨдәәжғҠиүізҡ„еҠҹиғҪгҖӮе®ғиғҪе°ҶжһҜзҮҘзҡ„ж–ҮжЎЈиҪ¬еҢ–дёәз”ҹеҠЁзҡ„еҜ№иҜқејҸйҹійў‘пјҡ

  • еҸҢдәәеҜ№иҜқз”ҹжҲҗпјҡжЁЎжӢҹдёӨдҪҚдё»жҢҒдәәзҡ„ж·ұеәҰи®Ёи®ә
  • иҮӘз„¶иҜӯйҹіеҗҲжҲҗпјҡеҢ…еҗ«еҒңйЎҝгҖҒиҜӯж°”гҖҒжғ…ж„ҹиЎЁиҫҫ
  • еҶ…е®№йҮҚжһ„пјҡдёҚжҳҜз®ҖеҚ•жң—иҜ»пјҢиҖҢжҳҜйҮҚж–°з»„з»Үе’Ңи§ЈйҮҠ
  • еҗ¬дј—еҜјеҗ‘пјҡдҪҝз”Ёзұ»жҜ”гҖҒжҸҗй—®гҖҒжҖ»з»“зӯүжүӢжі•

4. з»“жһ„еҢ–з¬”и®°з”ҹжҲҗ

  • FAQз”ҹжҲҗпјҡд»Һж–ҮжЎЈдёӯжҸҗеҸ–еёёи§Ғй—®йўҳ
  • еӯҰд№ жҢҮеҚ—пјҡеҲӣе»әз»“жһ„еҢ–зҡ„еӯҰд№ и·Ҝеҫ„
  • ж—¶й—ҙзәҝпјҡжҸҗеҸ–дәӢд»¶е№¶жҢүж—¶й—ҙжҺ’еәҸ
  • еӨ§зәІжҖ»з»“пјҡеӨҡеұӮзә§зҡ„еҶ…е®№жҰӮиҰҒ

class DocumentProcessor:
    """
    ж–ҮжЎЈеӨ„зҗҶеҷЁпјҡе°Ҷеҗ„з§Қж јејҸиҪ¬жҚўдёәз»ҹдёҖзҡ„з»“жһ„еҢ–иЎЁзӨә
    """
    def __init__(self):
        self.pdf_parser = PyMuPDFParser()
        self.html_parser = BeautifulSoupParser()
        self.audio_transcriber = WhisperTranscriber()
        self.vision_extractor = GeminiVisionExtractor()

    async def process_document(self, file_path: str, file_type: str) -> Document:
        """
        з»ҹдёҖзҡ„ж–ҮжЎЈеӨ„зҗҶжҺҘеҸЈ

        Args:
            file_path: ж–ҮжЎЈи·Ҝеҫ„
            file_type: ж–ҮжЎЈзұ»еһӢ (pdf, html, audio, etc.)

        Returns:
            з»“жһ„еҢ–зҡ„DocumentеҜ№иұЎ
        """
        # жӯҘйӘӨ1пјҡеҶ…е®№жҸҗеҸ–
        if file_type == "pdf":
            raw_content = self.pdf_parser.extract(file_path)
        elif file_type == "html":
            raw_content = self.html_parser.extract(file_path)
        elif file_type == "audio":
            raw_content = self.audio_transcriber.transcribe(file_path)

        # жӯҘйӘӨ2пјҡз»“жһ„еҢ–еӨ„зҗҶ
        structured_doc = self._structure_content(raw_content)

        # жӯҘйӘӨ3пјҡе…ғж•°жҚ®жҸҗеҸ–
        metadata = await self._extract_metadata(structured_doc)

        # жӯҘйӘӨ4пјҡеҲҶеқ—еӨ„зҗҶпјҲз”ЁдәҺembeddingпјү
        chunks = self._create_chunks(
            structured_doc,
            chunk_size=512,
            overlap=50,
            preserve_structure=True
        )

        return Document(
            content=structured_doc,
            metadata=metadata,
            chunks=chunks
        )

    def _create_chunks(self, doc: StructuredContent,
                       chunk_size: int, overlap: int,
                       preserve_structure: bool) -> List[Chunk]:
        """
        жҷәиғҪеҲҶеқ—пјҡдҝқжҢҒиҜӯд№үе®Ңж•ҙжҖ§

        е…ій”®зӮ№пјҡ
        - еңЁеҸҘеӯҗиҫ№з•ҢеҲҶеқ—
        - дҝқжҢҒз« иҠӮе®Ңж•ҙжҖ§
        - ж·»еҠ дёҠдёӢж–Үе…ғж•°жҚ®
        """
        chunks = []

        for section in doc.sections:
            # дҪҝз”Ёж»‘еҠЁзӘ—еҸЈпјҢдҪҶеңЁеҸҘеӯҗиҫ№з•Ң
            text = section.text
            sentences = self._split_sentences(text)

            current_chunk = []
            current_size = 0

            for sent in sentences:
                sent_tokens = len(self.tokenizer.encode(sent))

                if current_size + sent_tokens > chunk_size and current_chunk:
                    # еҲӣе»әchunkе№¶дҝқз•ҷдёҠдёӢж–ҮдҝЎжҒҜ
                    chunk_text = " ".join(current_chunk)
                    chunks.append(Chunk(
                        text=chunk_text,
                        metadata={
                            "section_title": section.title,
                            "doc_id": doc.id,
                            "position": len(chunks)
                        }
                    ))

                    # йҮҚеҸ еӨ„зҗҶпјҡдҝқз•ҷжңҖеҗҺеҮ дёӘеҸҘеӯҗ
                    overlap_sents = self._get_overlap_sentences(
                        current_chunk, overlap
                    )
                    current_chunk = overlap_sents + [sent]
                    current_size = sum(len(self.tokenizer.encode(s))
                                     for s in current_chunk)
                else:
                    current_chunk.append(sent)
                    current_size += sent_tokens

            # еӨ„зҗҶжңҖеҗҺдёҖдёӘchunk
            if current_chunk:
                chunks.append(Chunk(
                    text=" ".join(current_chunk),
                    metadata={
                        "section_title": section.title,
                        "doc_id": doc.id,
                        "position": len(chunks)
                    }
                ))

        return chunks

class RAGEngine:
    """
    RAGеј•ж“ҺпјҡжЈҖзҙўеўһејәз”ҹжҲҗ
    """
    def __init__(self, vector_store, llm_client, embedder):
        self.vector_store = vector_store
        self.llm = llm_client
        self.embedder = embedder
        self.reranker = CrossEncoderReranker()

    async def answer_query(self, query: str,
                          source_docs: List[str],
                          top_k: int = 5) -> Answer:
        """
        еҹәдәҺжәҗж–ҮжЎЈеӣһзӯ”й—®йўҳ

        жөҒзЁӢпјҡ
        1. жҹҘиҜўзҗҶи§Је’Ңжү©еұ•
        2. еҗ‘йҮҸжЈҖзҙў
        3. йҮҚжҺ’еәҸ
        4. дёҠдёӢж–Үжһ„е»ә
        5. LLMз”ҹжҲҗ
        6. зӯ”жЎҲйӘҢиҜҒ
        """
        # жӯҘйӘӨ1пјҡжҹҘиҜўеўһејә
        enhanced_query = await self._enhance_query(query)

        # жӯҘйӘӨ2пјҡж··еҗҲжЈҖзҙўпјҲеҗ‘йҮҸ+е…ій”®иҜҚпјү
        vector_results = await self._vector_search(
            enhanced_query,
            source_docs,
            top_k=top_k*2
        )
        keyword_results = await self._keyword_search(
            query,
            source_docs,
            top_k=top_k
        )

        # жӯҘйӘӨ3пјҡиһҚеҗҲе’ҢйҮҚжҺ’еәҸ
        combined = self._merge_results(vector_results, keyword_results)
        reranked = await self.reranker.rerank(query, combined, top_k=top_k)

        # жӯҘйӘӨ4пјҡжһ„е»әжҸҗзӨәиҜҚ
        prompt = self._build_rag_prompt(query, reranked)

        # жӯҘйӘӨ5пјҡLLMз”ҹжҲҗпјҲеёҰзәҰжқҹпјү
        response = await self.llm.generate(
            prompt,
            temperature=0.1,  # дҪҺжё©еәҰдҝқиҜҒеҮҶзЎ®жҖ§
            max_tokens=1024,
            stop_sequences=["[END]"]
        )

        # жӯҘйӘӨ6пјҡеј•з”ЁйӘҢиҜҒ
        validated_answer = self._validate_citations(
            response.text,
            reranked
        )

        return Answer(
            text=validated_answer.text,
            citations=validated_answer.citations,
            confidence=validated_answer.confidence
        )

    def _build_rag_prompt(self, query: str,
                          contexts: List[Chunk]) -> str:
        """
        жһ„е»әRAGжҸҗзӨәиҜҚ

        е…ій”®зәҰжқҹпјҡ
        - еҸӘдҪҝз”ЁжҸҗдҫӣзҡ„дёҠдёӢж–Ү
        - еҝ…йЎ»еј•з”ЁжқҘжәҗ
        - дёҚзЎ®е®ҡж—¶жҳҺзЎ®иҜҙжҳҺ
        """
        context_str = "\n\n".join([
            f"[Source {i+1}: {ctx.metadata['section_title']}]\n{ctx.text}"
            for i, ctx in enumerate(contexts)
        ])

        prompt = f"""дҪ жҳҜдёҖдёӘзІҫзЎ®зҡ„й—®зӯ”еҠ©жүӢпјҢеҸӘеҹәдәҺжҸҗдҫӣзҡ„ж–ҮжЎЈеҶ…е®№еӣһзӯ”й—®йўҳгҖӮ

дёҠдёӢж–Үж–ҮжЎЈпјҡ
{context_str}

з”ЁжҲ·й—®йўҳпјҡ{query}

еӣһзӯ”иҰҒжұӮпјҡ
1. еҸӘдҪҝз”ЁдёҠиҝ°ж–ҮжЎЈдёӯзҡ„дҝЎжҒҜ
2. еҜ№жҜҸдёӘдәӢе®һж ҮжіЁжқҘжәҗпјҲеҰӮ [Source 1]пјү
3. еҰӮжһңж–ҮжЎЈдёӯжІЎжңүзӣёе…ідҝЎжҒҜпјҢжҳҺзЎ®иҜҙжҳҺ"ж–ҮжЎЈдёӯжңӘжҸҗеҸҠ"
4. дёҚиҰҒж·»еҠ ж–ҮжЎЈеӨ–зҡ„зҹҘиҜҶ

еӣһзӯ”пјҡ"""

        return prompt

    async def _vector_search(self, query: str,
                            doc_ids: List[str],
                            top_k: int) -> List[Chunk]:
        """
        еҗ‘йҮҸжЈҖзҙў
        """
        # з”ҹжҲҗжҹҘиҜўеҗ‘йҮҸ
        query_embedding = await self.embedder.embed(query)

        # еңЁеҗ‘йҮҸж•°жҚ®еә“дёӯжЈҖзҙў
        results = await self.vector_store.search(
            embedding=query_embedding,
            filters={"doc_id": {"$in": doc_ids}},
            top_k=top_k,
            metric="cosine"
        )

        return results

    def _validate_citations(self, answer: str,
                           contexts: List[Chunk]) -> ValidatedAnswer:
        """
        йӘҢиҜҒзӯ”жЎҲдёӯзҡ„еј•з”ЁжҳҜеҗҰеҮҶзЎ®

        йҳІжӯўLLMзј–йҖ еј•з”Ё
        """
        cited_sources = self._extract_citations(answer)

        for citation in cited_sources:
            source_idx = citation.source_id - 1
            if source_idx >= len(contexts):
                # еј•з”ЁдәҶдёҚеӯҳеңЁзҡ„жқҘжәҗ
                answer = answer.replace(citation.text,
                                      f"{citation.text} [еј•з”ЁйӘҢиҜҒеӨұиҙҘ]")
            else:
                # йӘҢиҜҒеј•з”ЁеҶ…е®№жҳҜеҗҰеңЁеҺҹж–Үдёӯ
                source_text = contexts[source_idx].text
                if not self._fuzzy_match(citation.content, source_text):
                    answer = answer.replace(citation.text,
                                          f"{citation.text} [йңҖиҰҒдәәе·ҘйӘҢиҜҒ]")

        return ValidatedAnswer(text=answer, citations=cited_sources)

async def _synthesize_voices(self,
                            script: DialogueScript) -> List[AudioTrack]:
    """
    еҗҲжҲҗеҜ№иҜқиҜӯйҹі

        "Host": {
            "voice_id": "sarah_friendly",
            "pitch": 1.0,
            "speed": 1.05
        },
        "Expert": {
            "voice_id": "john_professional",
            "pitch": 0.95,
            "speed": 1.0
        }
    }

    for segment in script.segments:
        speaker = segment.speaker
        voice_config = voices[speaker]

        # еҗҲжҲҗиҜӯйҹіпјҲдҪҝз”Ёй«ҳиҙЁйҮҸTTSпјү
        audio = await self.tts.synthesize(
            text=segment.text,
            voice_id=voice_config["voice_id"],
            emotion=segment.emotion,
            speed=voice_config["speed"],
            pitch=voice_config["pitch"],
            add_breath=True,  # ж·»еҠ е‘јеҗёйҹі
            stability=0.5,    # йҖӮеәҰеҸҳеҢ–
            similarity_boost=0.75
        )

        # ж·»еҠ еҒңйЎҝ
        if segment.pause_after > 0:
            silence = self._generate_silence(segment.pause_after)
            audio = self._concatenate_audio([audio, silence])

        tracks.append(AudioTrack(
            audio=audio,
            speaker=speaker,
            timestamp=self._calculate_timestamp(tracks)
        ))

    return tracks

  • з”ЁжҲ·дҪ“йӘҢпјҡд»ҺеҜ№иҜқејҸй—®зӯ”еҲ°жІүжөёејҸж’ӯе®ўпјҢйҷҚдҪҺзҹҘиҜҶж¶Ҳиҙ№й—Ёж§ӣ
