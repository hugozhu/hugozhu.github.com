3.1 иҜ„дј°з»ҙеәҰи®ҫи®Ў

дёҖдёӘе®һз”Ёзҡ„ Judge Agent йңҖиҰҒд»ҺеӨҡдёӘз»ҙеәҰиҜ„дј°жү§иЎҢ Agent зҡ„е·ҘдҪңгҖӮд»ҘдёӢжҳҜдёҖдёӘе…ёеһӢзҡ„иҜ„дј°жЎҶжһ¶пјҡ

from dataclasses import dataclass, field
from typing import Optional
from enum import Enum


class ConfidenceLevel(Enum):
    """иҜ„дј°зҪ®дҝЎеәҰзӯүзә§"""
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"


@dataclass
class EvaluationDimension:
    """еҚ•дёӘиҜ„дј°з»ҙеәҰ"""
    name: str
    score: float  # 0.0 - 10.0
    weight: float  # жқғйҮҚ
    rationale: str  # иҜ„еҲҶзҗҶз”ұ
    evidence: Optional[str] = None  # иҜҒжҚ®еј•з”Ё


@dataclass
class JudgeVerdict:
    """Judge зҡ„жңҖз»ҲиЈҒеҶі"""
    overall_score: float
    dimensions: list[EvaluationDimension]
    confidence: ConfidenceLevel
    actionable_feedback: str
    critical_issues: list[str] = field(default_factory=list)
    suggestions: list[str] = field(default_factory=list)

    def is_passing(self, threshold: float = 6.0) -> bool:
        """жҳҜеҗҰйҖҡиҝҮиҜ„дј°"""
        return self.overall_score >= threshold

    def weighted_score(self) -> float:
        """и®Ўз®—еҠ жқғеҲҶж•°"""
        total_weight = sum(d.weight for d in self.dimensions)
        if total_weight == 0:
            return 0.0
        return sum(
            d.score * d.weight for d in self.dimensions
        ) / total_weight

from dataclasses import dataclass
from typing import Protocol


class ExecutionTrace(Protocol):
    """жү§иЎҢиҪЁиҝ№ вҖ” Agent зҡ„е·ҘдҪңи®°еҪ•"""

    def get_steps(self) -> list[dict]:
        """иҺ·еҸ–жүҖжңүжү§иЎҢжӯҘйӘӨ"""
        ...

    def get_intermediate_outputs(self) -> list[dict]:
        """иҺ·еҸ–дёӯй—ҙиҫ“еҮә"""
        ...

    def get_final_output(self) -> dict:
        """иҺ·еҸ–жңҖз»Ҳиҫ“еҮә"""
        ...


@dataclass
class JudgeAgent:
    """Agent-as-a-Judge ж ёеҝғе®һзҺ°"""

    evaluation_prompt_template: str
    dimensions: list[str]
    threshold: float = 6.0

    def evaluate(
        self,
        task_description: str,
        trace: ExecutionTrace,
    ) -> JudgeVerdict:
        """
        иҜ„дј°жү§иЎҢиҪЁиҝ№гҖӮ

        Args:
            task_description: еҺҹе§Ӣд»»еҠЎжҸҸиҝ°
            trace: Agent зҡ„жү§иЎҢиҪЁиҝ№

        Returns:
            JudgeVerdict: з»“жһ„еҢ–иҜ„дј°з»“жһң
        """
        # Step 1: и§ӮеҜҹ вҖ” ж”¶йӣҶжү§иЎҢиҜҒжҚ®
        observations = self._observe(trace)

        # Step 2: еҲҶжһҗ вҖ” йҖҗз»ҙеәҰиҜ„дј°
        dimensions = []
        for dim_name in self.dimensions:
            dim_eval = self._analyze_dimension(
                dim_name, task_description, observations
            )
            dimensions.append(dim_eval)

        # Step 3: иЈҒеҶі вҖ” з»јеҗҲеҲӨж–ӯ
        verdict = self._render_verdict(dimensions, observations)
        return verdict

    def _observe(self, trace: ExecutionTrace) -> dict:
        """и§ӮеҜҹжү§иЎҢиҪЁиҝ№пјҢжҸҗеҸ–е…ій”®дҝЎжҒҜ"""
        steps = trace.get_steps()
        outputs = trace.get_intermediate_outputs()
        final = trace.get_final_output()

        return {
            "step_count": len(steps),
            "key_decisions": [
                s for s in steps if s.get("type") == "decision"
            ],
            "intermediate_outputs": outputs,
            "final_output": final,
            "error_count": sum(
                1 for s in steps if s.get("type") == "error"
            ),
        }

    def _analyze_dimension(
        self,
        dimension: str,
        task: str,
        observations: dict,
    ) -> EvaluationDimension:
        """
        еҜ№еҚ•дёӘз»ҙеәҰиҝӣиЎҢиҜ„дј°гҖӮ

        е®һйҷ…е®һзҺ°дёӯпјҢиҝҷйҮҢдјҡи°ғз”Ё LLM жқҘз”ҹжҲҗ
        иҜ„еҲҶе’ҢзҗҶз”ұгҖӮ
        """
        # з®ҖеҢ–зӨәдҫӢ вҖ” е®һйҷ…еә”и°ғз”Ё LLM
        return EvaluationDimension(
            name=dimension,
            score=7.5,
            weight=1.0,
            rationale="зӨәдҫӢиҜ„дј°зҗҶз”ұ",
        )

    def _render_verdict(
        self,
        dimensions: list[EvaluationDimension],
        observations: dict,
    ) -> JudgeVerdict:
        """з»јеҗҲеҗ„з»ҙеәҰз”ҹжҲҗжңҖз»ҲиЈҒеҶі"""
        weighted = sum(
            d.score * d.weight for d in dimensions
        ) / sum(d.weight for d in dimensions)

        return JudgeVerdict(
            overall_score=round(weighted, 1),
            dimensions=dimensions,
            confidence=ConfidenceLevel.MEDIUM,
            actionable_feedback="з»јеҗҲиҜ„дј°еҸҚйҰҲ...",
        )

Agent-as-a-Judge и§ЈеҶізҡ„жҳҜдёҖдёӘиў«й•ҝжңҹеҝҪи§Ҷзҡ„й—®йўҳпјҡеҪ“ Agent иғҪе·ҘдҪң 6 е°Ҹж—¶гҖҒ24 е°Ҹж—¶з”ҡиҮіжӣҙд№…ж—¶пјҢи°ҒжқҘе‘ҠиҜүе®ғеҒҡеҫ—еҘҪдёҚеҘҪпјҹ

жІЎжңүиҝҷдёӘй—®йўҳзҡ„зӯ”жЎҲпјҢиҮӘиҝӣеҢ–е°ұж— д»Һи°Ҳиө·гҖӮе°ұеғҸдёҖдёӘдәәеҰӮжһңж°ёиҝңдёҚзҹҘйҒ“иҮӘе·ұеҒҡеҫ—еҘҪдёҚеҘҪпјҢе°ұдёҚеҸҜиғҪиҝӣжӯҘгҖӮ

иҜ„дј°еӣӣиұЎйҷҗжЁЎеһӢе‘ҠиҜүжҲ‘д»¬пјҡз¬¬еӣӣиұЎйҷҗпјҲй•ҝж—¶зЁӢ + й«ҳеӨҚжқӮеәҰпјүжҳҜеҪ“еүҚжңҖеӨ§зҡ„иҜ„дј°зңҹз©әеҢәгҖӮAgent-as-a-Judge дёҚжҳҜе®ҢзҫҺзҡ„и§ЈеҶіж–№жЎҲпјҢдҪҶе®ғжҳҜзӣ®еүҚжңҖеҠЎе®һзҡ„ж–№еҗ‘вҖ”вҖ”дёҚиҝҪжұӮ 100% зІҫеҮҶпјҢеҸӘиҰҒжңүж–№еҗ‘жҖ§пјҢе°ұиғҪй©ұеҠЁжҢҒз»ӯиҝӣеҢ–гҖӮ

2026 е№ҙиў«еҫҲеӨҡдәәз§°дёә Agent зҲҶеҸ‘е№ҙгҖӮдҪҶжҲ‘и®ӨдёәпјҢзңҹжӯЈеҶіе®ҡ Agent иғҪеҗҰд»Һ"иғҪз”Ё"иө°еҗ‘"еҘҪз”Ё"зҡ„пјҢдёҚжҳҜжЁЎеһӢжңүеӨҡејәпјҢиҖҢжҳҜиҜ„дј°й—ӯзҺҜжңүеӨҡе®Ңж•ҙгҖӮ
